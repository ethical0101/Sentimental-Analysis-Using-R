### Step 2: Install Required R Packages

Open RStudio and run this script:

```r
# =============================================================================
# PACKAGE INSTALLATION SCRIPT
# Run this once to install all required packages
# =============================================================================

# List of required packages
packages <- c(
  "tidyverse",      # Data manipulation
  "tm",             # Text mining
  "stringr",        # String operations
  "syuzhet",        # Emotion detection
  "caret",          # Machine learning
  "e1071",          # SVM & Naive Bayes
  "topicmodels",    # LDA topic modeling
  "ggplot2",        # Plotting
  "wordcloud",      # Word clouds
  "fmsb",           # Radar charts
  "shiny",          # Dashboard framework
  "shinydashboard", # Dashboard UI
  "DT",             # Interactive tables
  "plotly",         # Interactive plots
  "zoo",            # Time series
  "lubridate",      # Date handling
  "RColorBrewer",   # Colors
  "tidytext"        # Text analysis
)

# Function to install and load packages
install_and_load <- function(package_name) {
  if (!require(package_name, character.only = TRUE)) {
    cat(sprintf("Installing %s...\n", package_name))
    install.packages(package_name, dependencies = TRUE)
    library(package_name, character.only = TRUE)
    cat(sprintf("âś“ %s installed and loaded\n", package_name))
  } else {
    cat(sprintf("âś“ %s already installed\n", package_name))
  }
}

# Install all packages
cat("\n========================================\n")
cat("Installing Required Packages...\n")
cat("========================================\n\n")

for (pkg in packages) {
  tryCatch({
    install_and_load(pkg)
  }, error = function(e) {
    cat(sprintf("âś— Error installing %s: %s\n", pkg, e$message))
  })
}

cat("\n========================================\n")
cat("âś“ Installation Complete!\n")
cat("========================================\n\n")

# Verify installation
cat("Verifying installation...\n")
installed_count <- sum(sapply(packages, function(pkg) {
  require(pkg, character.only = TRUE, quietly = TRUE)
}))

cat(sprintf("\n%d out of %d packages successfully installed.\n", 
            installed_count, length(packages)))

if (installed_count == length(packages)) {
  cat("\nâś“ All packages ready! You can now run the analysis.\n")
} else {
  cat("\nâš  Some packages failed to install. Please check errors above.\n")
}
```

## đź”§ TROUBLESHOOTING

### Issue 1: Package Installation Fails

**Solution:**
```r
# Try installing from different mirror
options(repos = "https://cloud.r-project.org")

# Install specific package
install.packages("package_name", dependencies = TRUE)

# If still fails, install from GitHub
install.packages("devtools")
devtools::install_version("package_name", version = "x.x.x")
```

### Issue 2: "Cannot open file" Error

**Solution:**
```r
# Check current directory
getwd()

# Set correct directory
setwd("C:/correct/path/to/project")

# Use full path
results <- run_complete_analysis("C:/full/path/to/data.csv")
```

### Issue 3: Dashboard Won't Load

**Solution:**
```r
# Check if shiny is installed
library(shiny)

# Run dashboard with specific port
options(shiny.port = 8888)
source("dashboard_app.R")

# If browser doesn't open, manually go to:
# http://127.0.0.1:8888
```

### Issue 4: Out of Memory Error

**Solution:**
```r
# Increase memory limit (Windows)
memory.limit(size = 8000)  # 8GB

# Process data in chunks for large files
# Or reduce dataset size for testing
```

### Issue 5: UTF-8 Encoding Issues

**Solution:**
```r
# Read CSV with specific encoding
data <- read_csv("file.csv", locale = locale(encoding = "UTF-8"))

# Or try
data <- read.csv("file.csv", fileEncoding = "UTF-8", stringsAsFactors = FALSE)
```

### Issue 6: Visualization Files Not Created

**Solution:**
```r
# Check write permissions in directory
getwd()

# Manually create visualizations
source("visualization.R")
create_visualization_dashboard(results$data, 
                                results$ts_results$daily_sentiment,
                                results$ml_comparison)
```

## đź“Š CSV FORMAT REQUIREMENTS

### Minimum Required Format:


### Recommended Format:


### Date Formats Accepted:
- YYYY-MM-DD (recommended): 2026-02-01
- MM/DD/YYYY: 02/01/2026
- DD-MM-YYYY: 01-02-2026

### Common Issues:
